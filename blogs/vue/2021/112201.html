    # 事件监听v-on 简写@

    <!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8" />
    <title></title>
</head>
<body>
    <div id="app">
        <!-- 事件调用的方法没有参数()可以去掉 -->
        <button @click="btn1Clicke">按钮1</button>

        <!-- 事件定义时，写方法去掉了() 但方法本身需要一个参数时，这个时候Vue 会默认将浏览器生产的event事件对象作为参数传入方法 -->
        <button @click="btn2Clicke">按钮2</button>
