 * 微信的request
 */
function request(url, data = {}, method = "GET") {
  var contentType = 'application/json'
  return new Promise(function (resolve, reject) {
    wx.request({
      url: url,
      data: data,
      method: method,
      header: {
        'Content-Type': contentType
        // 'Authorization': 'Bearer ' + getDataByKey('token')
      },
      success: function (res) {
        let result = res.data;
        if (result.code == "0") {

          reject(result.message)
        } else {
          //请求失败
          reject("请求失败：" + result.message)
        }
      },
      fail: function (err) {
        //服务器连接异常
        reject("服务器连接异常，请检查网络再试")
      }
    })
  });
}

    # 获取openid

    // app.js
const api = require('/utils/request/api')
const req = require('/utils/request/request')
App({
  onLaunch() {
    // 展示本地存储能力
    const logs = wx.getStorageSync('logs') || []
    logs.unshift(Date.now())
    wx.setStorageSync('logs', logs)
    this.getUserInfo();
  },
  globalData: {
    userInfo: null,
    openid: ""
  },
  getUserInfo() {
    let app = this;
    // 登录
    wx.login({
      success: res => {
        // 发送 res.code 到后台换取 openId, sessionKey, unionId
        req.get(api.getopenid, {
            code: res.code
          })
          .then(res => {
            app.globalData.openid = res;
            if (this.userInfoReadyCallback) {
              this.userInfoReadyCallback(res)
            }
          }).catch(err => {
            console.log(err);
          })
      }
    })
  }
})

     public JsonResult Login(string code)
        {
            string openid = string.Empty;
            if (!string.IsNullOrEmpty(code))
            {
                string url = string.Format("https://api.weixin.qq.com/sns/jscode2session?appid={0}&secret={1}&js_code={2}&grant_type=authorization_code", AppID, AppSecret, code);
                HttpHelper http = new HttpHelper();
                HttpItem item = new HttpItem
                {
                    URL = url,
                    Method = "GET",
                    Timeout = 5000
                };
                HttpResult result = http.GetHtml(item);
                if (!result.Html.Contains("errmsg^"))
                {
                    JObject jo = JObject.Parse(result.Html);
                    if (jo.Property("openid") != null)
                    {
                        //正确流程，获取到openid;
                        return Json(new Results { code = "0", message = "获取openid成功", data = jo["openid"].ToString() }, JsonRequestBehavior.AllowGet);
                    }
                }
            }
            return Json(new Results { code = "0020", message = "获取openid失败" }, JsonRequestBehavior.AllowGet);
        }
